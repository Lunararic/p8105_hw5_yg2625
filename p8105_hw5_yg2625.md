### Tidy the resulting data

``` r
arm_tidy =
  arm_df %>% 
  select(-path) %>% 
  mutate(file_name = str_replace(file_name, ".csv", "")) %>% 
  separate(file_name, into = c("arm", "id"), sep = "_") %>% 
  gather(key = week, value = values, week_1:week_8) %>% 
  mutate(week = str_replace(week, "week_", ""))
```
